informalidad_sector <- data %>% 
  filter(PP04B_COD %in% c(0101:9900)) %>%

genero <- data %>% 
  group_by(CH04) %>% 
  summarise(Poblacion         = sum(PONDERA),
            Ocupados          = sum(PONDERA[ESTADO == 1]),

  select(-c(2:5)) %>%

Resulta interesante analizar en que sectores de actividad se encuentran empleados varones y mujeres, ya que existe una vasta literatura que establece que las mujeres tienen una ventaja comparativa en actividades de servicio mientras que los varones la tienen en actividades industriales. Estudiemos, entonces, como se distribuyen tanto varones como mujeres en los diferentes sectores de actividad.

ocupados_sector_genero<-data %>% 
  filter(ESTADO==1, PP04B_COD %in% c(0101:9900))%>%
  group_by(sector_actividad)%>%
  summarise(Ocupados_sector_v    =sum(PONDERA[CH04==1]),
            Ocupados_sector_m    =sum(PONDERA[CH04==2]))%>%
  mutate(Participacion_varones= Ocupados_sector_v/sum(Ocupados_sector_v),
         Participacion_mujeres= Ocupados_sector_m/sum(Ocupados_sector_m)) %>% 
  select(-c(2,3)) %>% 
  mutate(Participacion_varones=round(Participacion_varones,2),
         Participacion_mujeres=round(Participacion_mujeres,2))%>% 
  rename(Varones=Participacion_varones,
         Mujeres=Participacion_mujeres)%>%
  gather(.,
         key=Gnero,
         value=Participacin,
         2:3)

grafico2<-ggplot(ocupados_sector_genero, aes(x=reorder(sector_actividad,-Participacin),y=Participacin, fill=sector_actividad))+
  geom_bar(stat="identity", width=0.5)+
  labs(x = "Sector",
       y = "Participacin",
